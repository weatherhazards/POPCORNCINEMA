#1948 elections
country_event = {
  id = 8015801
  title = "United States $Year$ Presidential Election"
  desc = EVT_8015801_DESC
  picture = "nwo2_usa_elections_1948"
  
	fire_only_once = yes

	trigger = {
		tag = USA
		year = 1948 
		NOT = { has_global_flag = 1950_start_date_flag }
		NOT = { year = 1952 }
		election = no
		government = democracy
		NOT = { has_country_flag = elections_disabled }
		month = 11
	}

	mean_time_to_happen = { days = 8 modifier = { factor = 0.1 year = 1949 } }

	option = {
		name = "Harry S. Truman (Moderate Democrat)"
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = USA NOT = { OR = { pop_majority_ideology = big_tent pop_majority_ideology = liberal } } } } }
			owner = { add_country_modifier = { name = revolution_chaos duration = 1065 } }
		}
		
		#The president's party becomes the executive, ie the lower house in Vicky
		ruling_party_ideology = big_tent
		
		#Simulate the ideological divide
		any_pop = { 
			limit = {
				OR = { strata = rich type = farmers type = labourers culture = afro_american }
			}
			ideology = { factor = 0.2 value = big_tent }
		}
		
		ai_chance = {
			factor = 0.25
			modifier = {
				factor = 3
				OR = { pop_majority_ideology = big_tent pop_majority_ideology = liberal }
			}
			modifier = {
				factor = 0.3
				NOT = { OR = { pop_majority_ideology = big_tent pop_majority_ideology = liberal } }
			}
		}
	}
	
	option = {
		name = "Henry Wallace (Progressive Democrat)"
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = USA NOT = { OR = { pop_majority_ideology = progressive pop_majority_ideology = socialist } } } } }
			owner = { add_country_modifier = { name = revolution_chaos duration = 1065 } }
		}
		
		#The president's party becomes the executive, ie the lower house in Vicky
		ruling_party_ideology = progressive
		
		#Simulate the ideological divide
		any_pop = { 
			limit = {
				OR = { strata = poor culture = afro_american }
			}
			ideology = { factor = 0.2 value = progressive }
		}
		
		any_pop = {
			limit = { AND = { location = { is_core = CSA } culture = white_anglo } }
			scaled_consciousness = { factor = 10 ideology = populist }
			scaled_militancy = { factor = 10 ideology = populist }
			scaled_consciousness = { factor = 10 ideology = nationalist }
			scaled_militancy = { factor = 10 ideology = nationalist }
		}
		
		
		ai_chance = {
			factor = 0.25
			modifier = {
				factor = 3
				OR = { pop_majority_ideology = progressive pop_majority_ideology = socialist }
			}
			modifier = {
				factor = 0.3
				NOT = { OR = { pop_majority_ideology = progressive pop_majority_ideology = socialist } }
			}
		}
	}

	option = {
		name = "Thomas E. Dewey (Conservative Republican)"
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = USA NOT = { OR = { pop_majority_ideology = conservative pop_majority_ideology = traditionalist } } } } }
			owner = { add_country_modifier = { name = revolution_chaos duration = 1065 } }
		}
		
		#The president's party becomes the executive, ie the lower house in Vicky
		ruling_party_ideology = conservative
		
		#Simulate the ideological divide
		any_pop = {
			limit = { OR = { AND = { location = { is_core = CSA } NOT = { culture = afro_american } } strata = rich strata = middle } }
			ideology = {
				factor = 0.2

			factor = 0.25

				factor = 3

			}
			modifier = {
				factor = 0.3
				NOT = { OR = { pop_majority_ideology = conservative pop_majority_ideology = traditionalist } }
			}
		}
	}
	
	option = {
		name = "Strom Thurmond (Dixiecrat)"
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = USA NOT = { OR = { pop_majority_ideology = populist pop_majority_ideology = nationalist } } } } }

	fire_only_once = yes

	trigger = {
    	tag = USA
		year = 1963 NOT = { year = 1964 }

		OR = { 
			has_country_flag = jfk_elected

		}
	}

	option = {
		name = EVT_8015805_A

			}
			ideology = { factor = 0.1 value = progressive }
		}
		any_pop = { militancy = -2.0 consciousness = 1 }
	}
}
